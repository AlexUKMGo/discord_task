    function playBeep() {
        if (!ui.chkSound.checked) return;
        try {
            if (!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)();
            if (audioCtx.state === 'suspended') audioCtx.resume();

            const duration = parseFloat(ui.inpSound.value) || 1;
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();

            osc.type = 'sine';
            osc.frequency.value = 1000;

            osc.connect(gain);
            gain.connect(audioCtx.destination);

            osc.start();
            gain.gain.setValueAtTime(1, audioCtx.currentTime);
            gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + duration);
            osc.stop(audioCtx.currentTime + duration);
        } catch(e) {}
    }

    function startTimer() {
        if (isRunning) return;

        clearAllTimers();

        const t = parseInt(ui.inpTime.value);
        if (!t || t <= 0) return;

        // If fresh start or reset / 初次开始或重置后开始

            originalTime = t;
            timeLeft = t;

        isRunning = true;

        ui.circle.classList.remove('time-up-anim');

        updateVisuals();

        timerId = setInterval(() => {
            timeLeft--;
            updateVisuals();
            if (timeLeft <= 0) handleTimeUp();
        }, 1000);
    }

    // --- Time up handler / 时间到处理 ---
    function handleTimeUp() {
        clearAllTimers();

        // Red alert / 红色提示
        ui.progress.style.background = `#f44336`;
        ui.text.style.color = "#f44336";
        ui.circle.classList.add('time-up-anim');

        playBeep();

        let attemptSuccess = false;

        // Auto send / 自动发送（可选）
        if (ui.chkAutoSend && ui.chkAutoSend.checked) {
            if (ui.chkQueueMode.checked) {
                // Queue mode: shift a line / 队列模式：取一行
                if (messageQueue.length > 0) {
                    const nextMsg = messageQueue.shift();
                    updateQueueBtnText();
                    injectAndSend(nextMsg);
                    attemptSuccess = true;
                } else {
                    attemptSuccess = false;
                }
            } else {
                // Non-queue: if user is typing, do not force-send / 非队列：用户正在输入则不抢发
                const isTyping = (Date.now() - lastTypingTime) < 2000;
                if (!isTyping) attemptSuccess = triggerDiscordSend();
            }
        }

        /**
         * Critical logic / 关键逻辑
         * - If auto-send enabled but fails due to empty content:
         * - Loop OFF: stop timer.
         * - Loop ON : continue next round anyway.
         */
        if (ui.chkAutoSend.checked && !attemptSuccess) {
            if (!ui.chkLoop.checked) {
                isRunning = false;
                ui.btnStart.textContent = "开始 Start";
                ui.btnStart.classList.remove('btn-disabled');
                ui.inpTime.disabled = false;
                return;
            }
        }

        // Loop scheduling / 循环下一轮
        if (ui.chkLoop.checked) {
            ui.text.textContent = "完成 Done";
            const wait = (parseFloat(ui.inpSound.value) || 1) * 1000 + 500;
            loopTimeoutId = setTimeout(() => {
                if (ui.chkLoop.checked) {
                    resetTimer();
                    startTimer();
                }
            }, wait);
        } else {
            isRunning = false;
            ui.btnStart.textContent = "开始 Start";
            ui.btnStart.classList.remove('btn-disabled');
            ui.inpTime.disabled = false;
        }
    }

    function resetTimer() {
        clearAllTimers();

        isRunning = false;
        timeLeft = 0;
        originalTime = parseInt(ui.inpTime.value) || 60;

        ui.text.textContent = formatTime(0);
        ui.progress.style.background = `#2196F3`;
        ui.text.style.color = "#333";

        ui.inpTime.disabled = false;
    }

    function forceRestart() {
        resetTimer();
        setTimeout(() => startTimer(), 50);
    }

    // --- 6. Message injection / 消息注入与发送 ---
    function injectAndSend(text) {
        const inputBox = document.querySelector('div[role="textbox"][contenteditable="true"]');
        if (!inputBox) return;

        inputBox.focus();

        // Try paste event first / 优先模拟粘贴
        const clipboardData = new DataTransfer();
        clipboardData.setData('text/plain', text);
        const pasteEvent = new ClipboardEvent('paste', { clipboardData, bubbles: true, cancelable: true });
        inputBox.dispatchEvent(pasteEvent);

        // Fallback insertText / 兜底插入
        if (inputBox.textContent.trim() === '') {
            document.execCommand('insertText', false, text);
        }

        // Send with Enter / Enter发送
        setTimeout(() => {
            const enterEvent = new KeyboardEvent('keydown', {
                key: 'Enter', code: 'Enter', keyCode: 13, which: 13,
                bubbles: true, cancelable: true
            });
            inputBox.dispatchEvent(enterEvent);
        }, 100);
    }

    function triggerDiscordSend() {
        const inputBox = document.querySelector('div[role="textbox"][contenteditable="true"]');
        if (inputBox && inputBox.textContent.trim().length > 0) {
            inputBox.focus();
            const event = new KeyboardEvent('keydown', {
                key: 'Enter', code: 'Enter', keyCode: 13, which: 13,
                bubbles: true, cancelable: true
            });
            inputBox.dispatchEvent(event);
            return true;
        }
        return false;
    }

        ui.chkLiteMode.dispatchEvent(new Event('change'));
    });

    // Queue modal / 预输入弹窗
    ui.btnQueueOpen.addEventListener('click', () => {
        ui.queueInput.value = messageQueue.join('\n');
        ui.modal.style.display = 'flex';
        ui.overlay.style.display = 'block';
    });

    ui.btnQueueCancel.addEventListener('click', () => {
        ui.modal.style.display = 'none';
        ui.overlay.style.display = 'none';
    });

    ui.btnQueueSave.addEventListener('click', () => {
        messageQueue = ui.queueInput.value.split('\n').filter(line => line.trim() !== '');
        updateQueueBtnText();
        ui.modal.style.display = 'none';
        ui.overlay.style.display = 'none';
    });

    // --- 8. Network hooks / 发言检测（XHR + fetch） ---
    const isStrictSendMessage = (url, method) =>
        method.toUpperCase() === 'POST' && /\/channels\/\d+\/messages(\?|$)/.test(url);

    function hookNetwork() {
        const Win = typeof unsafeWindow !== 'undefined' ? unsafeWindow : window;

        // Hook XHR / 拦截XHR
        const origOpen = Win.XMLHttpRequest.prototype.open;
        Win.XMLHttpRequest.prototype.open = function(method, url) {
            this.addEventListener('load', function() {
                if (this.status >= 200 && this.status < 300 && isStrictSendMessage(url, method)) {
                    if (ui.chkAuto && ui.chkAuto.checked) forceRestart();
                }
            });
            return origOpen.apply(this, arguments);
        };

        // Hook fetch / 拦截fetch
        const origFetch = Win.fetch;
        Win.fetch = async function(...args) {
            const response = await origFetch(...args);
            const url = args[0] ? args[0].toString() : '';
            const opts = args[1] || {};
            if (response.ok && isStrictSendMessage(url, opts.method || 'GET')) {
                if (ui.chkAuto && ui.chkAuto.checked) forceRestart();
            }
            return response;
        };
    }

    // --- 9. Dragging / 拖拽 ---
    function onMouseDown(e) {
        if (e.target === ui.btnClose || e.target === ui.btnRestore || e.target === ui.btnHelp) return;

        // Normal mode: drag only header; Lite mode: drag anywhere (except resize corner)
        if (!ui.chkLiteMode.checked && !ui.header.contains(e.target)) return;

        const rect = ui.root.getBoundingClientRect();
        if (ui.chkLiteMode.checked && (e.clientX > rect.right - 15 && e.clientY > rect.bottom - 15)) return;

        isDragging = true;
        ui.root.classList.add('dragging');

        dragStartX = e.clientX;
        dragStartY = e.clientY;

        const computed = window.getComputedStyle(ui.root);
        initialLeft = parseInt(computed.left);
        initialTop = parseInt(computed.top);

        // If left is auto, compute from right / 如果left是auto，从right换算
        if (isNaN(initialLeft)) {
            initialLeft = window.innerWidth - parseInt(computed.right) - ui.root.offsetWidth;
            ui.root.style.left = initialLeft + 'px';
            ui.root.style.right = 'auto';
        }

        ui.root.style.cursor = ui.chkLiteMode.checked ? 'grabbing' : 'auto';
        if (!ui.chkLiteMode.checked) ui.header.style.cursor = 'grabbing';
    }

    function onMouseMove(e) {
        if (!isDragging) return;
        e.preventDefault();

        const dx = e.clientX - dragStartX;
        const dy = e.clientY - dragStartY;

        ui.root.style.left = `${initialLeft + dx}px`;
        ui.root.style.top = `${initialTop + dy}px`;
    }

    function onMouseUp(e) {
        if (!isDragging) return;

        isDragging = false;
        ui.root.classList.remove('dragging');

        ui.root.style.cursor = 'default';
        ui.header.style.cursor = 'move';

        // Lite mode click: start/reset if no drag / 挂件模式下，如果不是拖拽而是轻点，切换开始/重置
        const dist = Math.sqrt(Math.pow(e.clientX - dragStartX, 2) + Math.pow(e.clientY - dragStartY, 2));
        if (dist < 5 && ui.chkLiteMode.checked) {
            if (isRunning) resetTimer();
            else startTimer();
        }
    }

    ui.root.addEventListener('mousedown', onMouseDown);
    document.addEventListener('mousemove', onMouseMove);
    document.addEventListener('mouseup', onMouseUp);

    // Click circle in normal mode / 普通模式点击圆环开始/重置
    ui.circle.addEventListener('click', () => {
        if (!ui.chkLiteMode.checked) {
            isRunning ? resetTimer() : startTimer();
        }
    });

    ui.btnStart.addEventListener('click', startTimer);
    ui.btnReset.addEventListener('click', resetTimer);

    ui.btnClose.addEventListener('click', (e) => {
        e.stopPropagation();
        ui.root.style.display = 'none';
    });

    // Keep inside viewport on resize / 窗口变化保持在可视范围
    window.addEventListener('resize', () => {
        if (ui.root.style.display === 'none') return;

        const rect = ui.root.getBoundingClientRect();
        if (rect.right > window.innerWidth) ui.root.style.left = (window.innerWidth - rect.width - 20) + 'px';
        if (rect.bottom > window.innerHeight) ui.root.style.top = (window.innerHeight - rect.height - 20) + 'px';
        if (rect.top < 0) ui.root.style.top = '20px';
        if (rect.left < 0) ui.root.style.left = '20px';
    });

    // Tampermonkey menu / 菜单
    if (typeof GM_registerMenuCommand !== 'undefined') {
        GM_registerMenuCommand("🔄 打开/重置位置 Open & Reset Position", () => {
            ui.root.style.display = 'flex';
            ui.root.style.top = '20px';
            ui.root.style.right = '20px';
            ui.root.style.left = 'auto';
            alert('倒计时已归位！Timer reset to default position!');
        });
    }

    // Init / 初始化
    updateQueueBtnText();
    updateVisuals();
    hookNetwork();
})();
